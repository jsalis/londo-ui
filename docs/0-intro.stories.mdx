### Installation

```bash
npm install londo-ui styled-components
```

### Usage

```jsx
import React from "react";
import { ThemeProvider, ResetStyle, NumberInput } from "londo-ui";

function App() {
    return (
        <ThemeProvider theme="dark">
            <ResetStyle />
            <NumberInput />
        </ThemeProvider>
    );
}
```
